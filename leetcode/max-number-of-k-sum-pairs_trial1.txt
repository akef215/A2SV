class Solution:
    def maxOperations(self, nums: List[int], k: int) -> int:
        nums.sort()
        out = 0
        i, j = 0, len(nums) - 1
        while j > i:
            s = nums[i] + nums[j]
            if s == k:
                out += 1
                i += 1
                j -= 1
            elif s > k:
                j -= 1
            else:
                i += 1
        return out
